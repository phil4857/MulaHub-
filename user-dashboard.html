const BACKEND_URL = "https://repo-1red-jipate-bonus-1.onrender.com"; document.addEventListener('DOMContentLoaded', () => { const username = localStorage.getItem("username"); if (!username) { window.location.href = "login.html"; return; } const balanceEl = document.getElementById("balance"); const earningsEl = document.getElementById("earnings"); const msgEl = document.getElementById("msg"); const amountInput = document.getElementById("amount"); async function loadDashboard() { try { const res = await fetch(`\( {BACKEND_URL}/dashboard?username= \){encodeURIComponent(username)}`); if (!res.ok) { const data = await res.json().catch(() => ({})); if (res.status === 403) { msgEl.style.color = "red"; msgEl.textContent = data.detail || "Account not approved yet."; } else { msgEl.style.color = "red"; msgEl.textContent = data.detail || `Error ${res.status}`; } return; } const data = await res.json(); if (balanceEl) { balanceEl.textContent = Number(data.balance || 0).toFixed(2); } if (earningsEl) { earningsEl.textContent = Number(data.earnings || 0).toFixed(2); } // Optional: show total invested let invested = 0; if (data.investments && typeof data.investments === 'object') { Object.values(data.investments).forEach(item => { invested += Number(item.amount || 0); }); } const investedEl = document.getElementById("invested"); if (investedEl) { investedEl.textContent = invested.toFixed(2); } msgEl.textContent = ""; } catch (err) { console.error("Dashboard load error:", err); msgEl.style.color = "red"; msgEl.textContent = "Cannot connect to server. Try again later."; } } async function requestWithdrawal() { const amount = parseFloat(amountInput?.value || 0); if (!amount || amount <= 0) { msgEl.style.color = "red"; msgEl.textContent = "Please enter a valid amount greater than 0."; return; } msgEl.textContent = "Processing..."; msgEl.style.color = "blue"; const formData = new URLSearchParams(); formData.append("username", username); formData.append("amount", amount); try { const res = await fetch(`${BACKEND_URL}/withdraw/request`, { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: formData }); const data = await res.json(); if (res.ok) { msgEl.style.color = "green"; msgEl.textContent = data.message || "Withdrawal request sent. Waiting for admin approval."; amountInput.value = ""; // clear input } else { msgEl.style.color = "red"; msgEl.textContent = data.detail || "Request failed."; } } catch (err) { msgEl.style.color = "red"; msgEl.textContent = "Network error. Please check connection."; console.error(err); } } function logout() { localStorage.removeItem("username"); window.location.href = "login.html"; } // Attach events document.querySelector("button[onclick='logout()']")?.addEventListener("click", logout); document.querySelector("button[onclick='withdraw()']")?.addEventListener("click", requestWithdrawal); // Load data loadDashboard(); });
